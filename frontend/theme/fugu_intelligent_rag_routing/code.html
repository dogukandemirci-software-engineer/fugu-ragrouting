import { FuguClient } from '@fugu/sdk';

const fugu = new FuguClient({
  apiKey: process.env.FUGU_API_KEY

// Automatic hybrid routing between vector & graph

  prompt: "Who manages the team that deployed the auth service?",
  strategy: "auto", // Let Fugu decide
  options: {
    explain: true
